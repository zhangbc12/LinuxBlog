动态规划由浅入深

动态规划解题基本步骤

1. 定义数组的含义
2. 明确动态规划转移方程
3. 找到初始值


动态规划问题分类


不同路径


一个机器人位于一个 m x n 网格的左上角。机器人每次只能向下或者向右移动一步。机器人试图达到网格的右下角。问总共有多少条不同的路径？

方法：定义dp[i][j]表示机器人在走到网格i, j时所有可能的方法数，其中i = 0 or j = 0时方法数都为1。

状态转移方程为:

dp[i][j] = dp[i-1][j] + dp[i][j-1] (i > 0, j > 0)

// 2d vector version
int uniquePaths(int m, int n) {
   vector<vector<int>> dp(m, vector<int>(n, 0));
   for (int i = 0; i < m; ++i) {
      dp[i][0] = 1;
   }
   for (int j = 0; j < n; ++j) {
      dp[0][j] = 1;
   }
   for(int i = 1; i < m; i++) {
      for(int j = 1; j < n; j++) {
         dp[i][j] = dp[i - 1][j] + dp[i][j - 1];
      }
   }
   return dp[m - 1][n - 1];
}
// Time Complexity: O(N*M) 
// Auxiliary Space: O(N*M)

可以把dp压缩到一维数组

// simplified 1d vector version
int uniquePaths(int m, int n) {
   vector<int> dp(n, 0);
   dp[0] = 1;
   for(int i = 0; i < m; i++) {
      for(int j = 1; j < n; j++) {
         dp[j] += dp[j - 1];
      }
   }
   return dp[n - 1];
}
// Time Complexity: O(N*M) 
// Auxiliary Space: O(N)

不同路径(障碍)


一个机器人位于一个 m x n 网格的左上角。机器人每次只能向下或者向右移动一步。机器人试图达到网格的右下角。现在考虑网格中有障碍物。那么从左上角到右下角将会有多少条不同的路径？

方法：定义dp[i][j]表示机器人在走到网格i, j时所有可能的方法数，其中i = 0 or j = 0时，需要对dp[0][j]和dp[i][0]这两条路径的初始状态进行定义，在从dp[0][0]一种向下或者一直向右走的时候，如果遇到障碍，则后面的点都无法从dp[0][0]到达。

初始状态定义为：

for (int i = 0; i < m && obstacleGrid[i][0] == 0; i++) dp[i][0] = 1;
for (int j = 0; j < n && obstacleGrid[0][j] == 0; j++) dp[0][j] = 1;

在遍历的过程中，如果该点有障碍，则方法数为0，无障碍则按照状态转移方程遍历:

if(obstacleGrid[i][j] == 0) dp[i][j] = dp[i - 1][j] + dp[i][j - 1]

// 2d vector version
int uniquePathsWithObstacles(vector<vector<int>>& obstacleGrid) {
   int m = obstacleGrid.size();
   int n = obstacleGrid[0].size();
   vector<vector<int>> dp(m, vector(n, 0));
   for(int i = 0; i < m; i++) {
         if(obstacleGrid[i][0] == 1) break;
         dp[i][0] = 1;
   }
   for(int i = 0; i < n; i++) {
         if(obstacleGrid[0][i] == 1) break;
         dp[0][i] = 1;
   }

   for(int i = 1; i < m; i++) {
         for(int j = 1; j < n; j++) {
            if(obstacleGrid[i][j] == 0) {
               dp[i][j] = dp[i - 1][j] + dp[i][j - 1];
            }
         }
   }
   return dp[m - 1][n - 1];
}
// Time Complexity: O(N*M) 
// Auxiliary Space: O(N*M)

可以把dp压缩到一维数组

// simplified 1d vector version
int uniquePathsWithObstacles(vector<vector<int>>& obstacleGrid) {
   int m = obstacleGrid.size();
   int n = obstacleGrid[0].size();
   vector<int> dp(n);
   dp[0] = obstacleGrid[0][0] == 0 ? 1: 0;

   for(int i = 0; i < m; i++) {
         for(int j = 0; j < n; j++) {
            if(obstacleGrid[i][j] == 1) {
               dp[j] = 0;
            }
            else if(j > 0) {
               dp[j] += dp[j - 1];
            }
         }
   }
   return dp[n - 1];
}
// Time Complexity: O(N*M) 
// Auxiliary Space: O(N)

01背包(01 Knapsack Problem)


01的意思是，每种物品只会放入背包0个或者一个。假设一共有N件物品，第i件物品的重量为weights[i]，价值为values[i]。在总重量不超过背包承载上限W的情况下，此时能够装入背包的最大价值是多少？

方法：定义dp[i][j]为将前i件物品装进限重为j的背包可以获得的最大价值，在没有装入物品时的最大价值为0，在考虑i时，根据目前的重量j，有可能装入和不可能装入两种情况：

1. 如果weights[i] <= j, 可能能装入，状态转移方程为:

dp[i][j] = max(dp[i - 1][j], dp[i - 1][j - weights[i - 1]] + values[i - 1])

2. 如果weights[i] > j, 不可能装入，状态转移方程为:

dp[i][j] = dp[i][j - 1]

// 2d vector version
int zero_one_package(vector<int> weights, vector<int> values, int w) {
   int n = weights.size();
   vector<vector<int>> dp(n + 1, vector<int>(w + 1));
   for(int i = 0; i < n; i++) {
      for(int j = 0; j < w; j++) {
         if(i == 0 || j == 0) {
            dp[i][j] == 0;
         }
         else if(weights[i - 1] <= j) {
            dp[i][j] = max(dp[i - 1][j], dp[i - 1][j - weights[i - 1]] + values[i - 1]);
         }
         else {
            dp[i][j] = dp[i - 1][j];
         }
      }
   }
   return dp[n][w];
}
// Time Complexity: O(N*W) 
// Auxiliary Space: O(N*W)

可以把dp压缩到一维数组，需要在循环j时进行逆向枚举。

// simplified 1d vector version
int zero_one_package(vector<int> weights, vector<int> values, int w) {
   int n = weights.size();
   vector<int> dp(w + 1， 0);
   for(int i = 0; i < n; i++) {
      for(int j = w - 1; j >= weights[i]; j--) {
         dp[j] = max(dp[j], dp[j - weights[i]] + values[i]);
      }
   }
   return dp[w];
}
// Time Complexity: O(N*W) 
// Auxiliary Space: O(W)

无限背包问题(Unbounded Knapsack Problem)


无限背包问题于01背包问题类型相似，不同之处在于物品有许多种类，每种物品都无限量供应。

方法一：仿照01背包，同样地去定义dp[i][j],不同在于可以多次装入一种物品，动态转移方程为：

if(j >= weights[i - 1]) dp[i][j] = max(dp[i - 1][j], dp[i][j - weights[i - 1]] + values[i - 1])

else dp[i][j] = dp[i - 1][j]

// 2d vector version
int zero_one_package(vector<int> weights, vector<int> values, int w) {
   int n = weights.size();
   vector<vector<int>> dp(n + 1, vector<int>(w + 1));
   for(int i = 0; i < n; i++) {
      for(int j = 0; j < w; j++) {
         if(i == 0 || j == 0) {
            dp[i][j] == 0;
         }
         else if(weights[i - 1] <= j) {
            dp[i][j] = max(dp[i - 1][j], dp[i][j - weights[i - 1]] + values[i - 1]);
         }
         else {
            dp[i][j] = dp[i - 1][j];
         }
      }
   }
   return dp[n][w];
}
// Time Complexity: O(N*W) 
// Auxiliary Space: O(N*W)

可以把dp压缩到一维数组，需要在循环j时进行正向枚举。

// simplified 1d vector version
int zero_one_package(vector<int> weights, vector<int> values, int w) {
   int n = weights.size();
   vector<int> dp(w + 1， 0);
   for(int i = 0; i < n; i++) {
      for(int j = weights[i]; j < w; j++) {
         dp[j] = max(dp[j], dp[j - weights[i]] + values[i]);
      }
   }
   return dp[w];
}
// Time Complexity: O(N*W) 
// Auxiliary Space: O(W)

方法二：仿照01背包，同样地去定义dp[i][j],去逆向枚举，并考虑每种取物品的数量x∈[0, j//weights[i]]，动态转移方程为：

dp[i][j] = max(dp[i - 1][j - x*weights[i]] + x*values[i])

// simplified 1d vector version
   int zero_one_package(vector<int> weights, vector<int> values, int w) {
      int n = weights.size();
      vector<int> dp(w + 1， 0);
      for(int i = 0; i < n; i++) {
         for(int j = w - 1; j >= weights[i]; j++) {
            int x = j / weights[i];
            for(int k = 0; k < x; k++) {
               dp[j] = max(dp[j], dp[j - k * weights[i]] + k * values[i]);
            }
         }
      }
      return dp[w];
   }
   // Time Complexity: O(N*W*X) 
   // Auxiliary Space: O(W)
